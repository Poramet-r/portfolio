def clean_and_update_textbox(self):
    raw_text = self.name_entry.get("1.0", tk.END).strip()
    text = re.sub(r"\)(?=[^\s\n])", ")\n", raw_text)

    final_names = []
    for line in text.split("\n"):
        clean_line = re.sub(r"\(.*?\)", "", line).strip()
        clean_line = re.sub(r"^(Mr\.|Ms\.|Mrs\.)\s*", "", clean_line)
        if clean_line:
            final_names.append(clean_line)

    return final_names

def build_comparison(real_rows, sub_rows):
    sub_by_out = {s["out_id"]: s for s in sub_rows if s["out_id"]}
    sub_by_pos = defaultdict(list)

    for s in sub_rows:
        sub_by_pos[s["pos_key"]].append(s)

    result = []
    for r in real_rows:
        s = sub_by_out.get(r["out_id"])
        if not s:
            for key in [r["pos_display"], r["pos_base"]]:
                if sub_by_pos.get(key):
                    s = sub_by_pos[key][0]
                    break

        diffs = compare_fields(r, s) # supervisor, outsource, wage, plan
        result.append({**r, "found": s is not None, "diffs": diffs})
    return result

KPI Auto-Collect System

An automated KPI collection system built on Google Apps Script and LINE Bot. Employees reply with their AL Code directly in LINE — the system validates, records the response to Google Sheets in real time, and generates a monthly report with no manual data entry required.

03
GAS pattern matchingThe script validates the AL Code format. Unrecognised messages are passed to Gemini AI to attempt a parse before rejecting.

Tech stack

  • Google Apps Script — backend & webhook handler
  • LINE Messaging API — employee-facing chat interface
  • Google Sheets — live database & reporting layer
  • Gemini AI API — fuzzy-parse messages that don't match the expected pattern

Design choice

LINE was chosen as the interface because every employee already has it installed, eliminating app adoption friction entirely.

Simplified GAS webhook handler — internal endpoints and sheet IDs are masked.

function doPost(e) {
  const body = JSON.parse(e.postData.contents);
  const events = body.events;

  events.forEach(function(event) {
    if (event.type !== "message") return;

    const text = event.message.text.trim().toUpperCase();
    const userId = event.source.userId;

    if (isValidALCode(text)) {
      saveToSheet(userId, text, new Date());
      replyMessage(event.replyToken, "✅ Recorded: " + text);
    } else {
      // Fallback: ask Gemini to parse ambiguous input
      const parsed = geminiParseAL(text);
      if (parsed) saveToSheet(userId, parsed, new Date());
    }
  });
}

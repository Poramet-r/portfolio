function doPost(e) {
  var data  = JSON.parse(e.postData.contents);
  var event = data.events[0];
  var text  = event.message.text;

  // จับ Pattern AL + slash commands
  var match = getMatchPattern(text);
  if (!match || !quotedMessageId) return;

  // บันทึกลง Sheet ตาม type
  if (match.startsWith("AL")) {
    extractSheet.appendRow([dateExcel, match, originalText, "", senderName]);
  } else {
    allCaseSheet.appendRow([originalTimestamp, match, originalText]);
  }
}

class Campaign(db.Model):
    # 1 ตำแหน่งงาน = 1 Campaign
    id       = db.Column(db.Integer, primary_key=True)
    name     = db.Column(db.String(200))  # ชื่อตำแหน่ง
    color    = db.Column(db.String(20))   # สีแคมเปญ
    groups   = db.relationship('Group',        backref='campaign')
    templates= db.relationship('PostTemplate', backref='campaign')

class PostLog(db.Model):
    # ติดตามสถานะโพสต์ทุกรายการ
    status       = db.Column(db.String(20), default='pending')
    scheduled_at = db.Column(db.DateTime)
    posted_at    = db.Column(db.DateTime, nullable=True)
    error_message= db.Column(db.Text, nullable=True)

    # แกะ HTML table → DataFrame
